class EvaluacionDiagnostica:
    """
    Sistema que eval√∫a el nivel REAL del estudiante mediante ejercicios adaptativos
    """
    
    def evaluar_matematicas(self, estudiante_id):
        """
        5 ejercicios adaptativos de matem√°ticas
        - Empieza con nivel intermedio
        - Si acierta: sube dificultad
        - Si falla: baja dificultad
        """
        nivel_actual = 'intermedio'
        aciertos = 0
        
        for i in range(5):
            ejercicio = self.generar_ejercicio(nivel_actual)
            respuesta = self.obtener_respuesta(estudiante_id, ejercicio)
            
            if respuesta.es_correcta:
                aciertos += 1
                if nivel_actual == 'basico':
                    nivel_actual = 'intermedio'
                elif nivel_actual == 'intermedio':
                    nivel_actual = 'avanzado'
            else:
                if nivel_actual == 'avanzado':
                    nivel_actual = 'intermedio'
                elif nivel_actual == 'intermedio':
                    nivel_actual = 'basico'
        
        # Clasificaci√≥n final basada en aciertos
        if aciertos >= 4:
            return 'avanzado'
        elif aciertos >= 2:
            return 'intermedio'
        else:
            return 'basico'
    
    def evaluar_lectura(self, estudiante_id):

        Evaluaci√≥n de comprensi√≥n lectora adaptativa
        """

        texto = self.seleccionar_texto_por_grado()
        
        # Medir velocidad de lectura

        respuestas = self.hacer_preguntas_comprension(estudiante_id, texto)
        
        # Clasificar

class SistemaObservacional:
    """
    Refina el perfil observando comportamiento real en la plataforma
    """
    
    def analizar_atencion_real(self, estudiante_id, sesion_id):
        """
        Mide atenci√≥n real bas√°ndose en:
        - Tiempo en cada ejercicio
        - Clics fuera de la tarea
        - Pausas frecuentes
        """
        metricas = self.obtener_metricas_sesion(sesion_id)
        
        tiempo_promedio_tarea = metricas['tiempo_total'] / metricas['num_tareas']
        tasa_distraccion = metricas['clics_fuera'] / metricas['tiempo_total']
        
        if tiempo_promedio_tarea > 25 and tasa_distraccion < 0.1:
            return 'alta'
        elif tiempo_promedio_tarea > 10:
            return 'media'
        else:
            return 'baja'
    
    def analizar_frustracion_real(self, estudiante_id):
        """
        Detecta frustraci√≥n por:
        - Abandonos tras errores
        - Tiempo de inactividad despu√©s de fallar
        - Tasa de rendici√≥n
        """
        historial = self.obtener_historial_intentos(estudiante_id)
        
        abandonos_post_error = 0
        intentos_persistentes = 0
        
        for intento in historial:
            if not intento.correcto:
                if intento.siguiente_accion == 'abandonar':
                    abandonos_post_error += 1
                elif intento.siguiente_accion == 'reintentar':
                    intentos_persistentes += 1
        
        ratio_abandono = abandonos_post_error / len(historial)
        
        if ratio_abandono < 0.2:
            return 'resiliente'
        elif ratio_abandono < 0.5:
            return 'intermedio'
        else:
            return 'sensible'
    
    def refinar_perfil_completo(self, estudiante_id, sesiones=5):
        """
        Despu√©s de 5 sesiones, refina el perfil inicial con datos reales
        """
        perfil_inicial = self.obtener_perfil_inicial(estudiante_id)
        perfil_refinado = perfil_inicial.copy()
        
        # Actualizar con observaciones reales
        perfil_refinado['atencion'] = self.analizar_atencion_real(estudiante_id, sesiones)
        perfil_refinado['frustracion'] = self.analizar_frustracion_real(estudiante_id)
        perfil_refinado['nivel_matematicas'] = self.calcular_nivel_real_matematicas(estudiante_id)
        
        # Detectar estilo de aprendizaje por preferencias reales
        perfil_refinado['estilo_aprendizaje'] = self.detectar_estilo_preferido(estudiante_id)
        
        return perfil_refinado
